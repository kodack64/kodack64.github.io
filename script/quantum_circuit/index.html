    • ベル状態生成

    <div id="bell"></div>
<script>
const bell = {
    operations: [
        {name: "H", step: 0, target: [0], color_fill: "#eeeeee", color_stroke: "#ee0000",}, 
        {name: "X", control: [0], step: 1, target: [1] }, 
    ]
}
const element = document.getElementById("bell");
const obj = new QuantumCircuitDrawerTwo(element, bell);
obj.draw();
</script>
